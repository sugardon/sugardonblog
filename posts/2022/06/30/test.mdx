  const match = /language-(\w+)/.exec(p.className || "");

  return match ? (
    <SyntaxHighlighter language={match[1]}>{p.children}</SyntaxHighlighter>
  ) : (
    <SyntaxHighlighter>{p.children}</SyntaxHighlighter>
  );
};
